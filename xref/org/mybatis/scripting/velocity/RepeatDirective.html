38  /**
39   * #repeat($collection $item SEP OPEN CLOSE).
40   */
41  public class RepeatDirective extends Directive {

44    private String var;

46    private String close = "";

52      return "repeat";

55    @Override
56    public void init(RuntimeServices rs, InternalContextAdapter context, Node node) throws TemplateInitException {
57      super.init(rs, context, node);
58      final int n = node.jjtGetNumChildren() - 1;
59      for (int i = 1; i < n; i++) {
60        Node child = node.jjtGetChild(i);
61        if (i == 1) {
62          if (child.getType() == ParserTreeConstants.JJTREFERENCE) {
63            var = ((ASTReference) child).getRootString();
64          } else {
65            throw new TemplateInitException("Syntax error", getTemplateName(), getLine(), getColumn());
66          }
67        } else if (child.getType() == ParserTreeConstants.JJTSTRINGLITERAL) {
68          String value = (String) ((ASTStringLiteral)child).value(context);
69          switch (i) {
70            case 2:
71              separator = value;
72              break;
73            case 3:
74              open = value;
75              break;
76            case 4:
77              close = value;
78              break;
79          }
80        } else {
81          throw new TemplateInitException("Syntax error", getTemplateName(), getLine(), getColumn());
82        }
83      }
84      uberInfo = new Info(this.getTemplateName(), getLine(), getColumn());
85    }

87    @Override
88    public boolean render(InternalContextAdapter context, Writer writer, Node node)
89        throws IOException, ResourceNotFoundException, ParseErrorException, MethodInvocationException {
90  
91      writer.append(open);
92  
93      Object listObject = node.jjtGetChild(0).value(context);
94  
95      if (listObject == null) {
96        return false;
97      }
98  
99      Iterator i = null;
100 
101     try {
102       i = rsvc.getUberspect().getIterator(listObject, uberInfo);
103     } catch (RuntimeException e) {
104       throw e;
105     } catch (Exception ee) {
106       String msg = "Error getting iterator for #repeat at " + uberInfo;
107       rsvc.getLog().error(msg, ee);
108       throw new VelocityException(msg, ee);
109     }
110 
111     if (i == null) {
112       throw new VelocityException("Invalid collection");
113     }
114 
115     int counter = 0;
116     boolean maxNbrLoopsExceeded = false;
117     Object o = context.get(var);
118 
119     ParameterMappingCollector collector = (ParameterMappingCollector) context.get(SQLScriptSource.MAPPING_COLLECTOR_KEY);
120     String savedItemKey = collector.getItemKey();
121     collector.setItemKey(var);
122     RepeatScope foreach = new RepeatScope(this, context.get(getName()), var);
123     context.put(getName(), foreach);
124 
125     NullHolderContext nullHolderContext = null;
126 
127     while (!maxNbrLoopsExceeded && i.hasNext()) {
128       Object value = i.next();
129       put(context, var, value);
130       foreach.index++;
131       foreach.hasNext = i.hasNext();
132 
133       try {
134         if (value == null) {
135           if (nullHolderContext == null) {
136             nullHolderContext = new NullHolderContext(var, context);
137           }
138           node.jjtGetChild(node.jjtGetNumChildren()-1).render(nullHolderContext, writer);
139         } else {
140           node.jjtGetChild(node.jjtGetNumChildren()-1).render(context, writer);
141         }
142       } catch (StopCommand stop) {
143         if (stop.isFor(this)) {
144           break;
145         } else {
146           clean(context, o, collector, savedItemKey);
147           throw stop;
148         }
149       }
150 
151       if (i.hasNext()) {
152         writer.append(separator);
153       }
154 
155       counter++;
156       maxNbrLoopsExceeded = counter >= 1000;
157     }
158     writer.append(close);
159     clean(context, o, collector, savedItemKey);
160     return true;
161 
162   }

164   protected void clean(InternalContextAdapter context,
165       Object o, ParameterMappingCollector collector, String savedItemKey) {
166     if (o != null) {
167       context.put(var, o);
168     } else {
169       context.remove(var);
170     }
171     collector.setItemKey(savedItemKey);
172     postRender(context);
173   }

233   protected static class NullHolderContext extends ChainedInternalContextAdapter {
234 
235     private String loopVariableKey = "";
236     private boolean active = true;
237 
238     private NullHolderContext(String key, InternalContextAdapter context) {
239       super(context);
240       if (key != null) {
241         loopVariableKey = key;
242       }
243     }
244 
245     @Override
246     public Object get(String key) throws MethodInvocationException {
247       return (active && loopVariableKey.equals(key))
248           ? null
249           : super.get(key);
250     }
251 
252     @Override
253     public Object put(String key, Object value) {
254       if (loopVariableKey.equals(key) && (value == null)) {
255         active = true;
256       }
257 
258       return super.put(key, value);
259     }
260 
261     @Override
262     public Object localPut(final String key, final Object value) {
263       return put(key, value);
264     }
265 
266     @Override
267     public Object remove(Object key) {
268       if (loopVariableKey.equals(key)) {
269         active = false;
270       }
271       return super.remove(key);
272     }
273   }

275   @Override
276   public boolean isScopeProvided() {
277     return true;
